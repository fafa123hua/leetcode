var searchInsert = function(nums, target) {
    for(let i = 0; i < nums.length; i++){
        if(target < nums[i]){
            if(i == 0){
                return 0
            }else{
                return i
            }
        }
        if(nums[i] == target){
            return i
        }
        if(i + 1 == nums.length){
            return i + 1
        }
    }
};
