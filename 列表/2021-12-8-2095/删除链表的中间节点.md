和867题差不多，额外注意一个!head.next为空的情况

```js
var deleteMiddle = function(head) {
    if(!head.next){
        return null
    }
    let s = q = head
    let ans
    while(q && q.next){      
        ans = s
        s = s.next
        q = q.next.next
    }
    ans.next = ans.next.next
    return head
};
```
