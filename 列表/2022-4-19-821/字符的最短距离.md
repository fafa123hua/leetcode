var shortestToChar = function(s, c) {
    let ans = new Array(s.length).fill(0)
    for (let i = 0, idx = -s.length; i < s.length; ++i) {
        if (s[i] === c) {
            idx = i;
        }
        ans[i] = i - idx;
    }
    for (let i = s.length - 1, idx = 2 * s.length; i >= 0; --i) {
        if (s[i] == c) {
            idx = i;
        }
        ans[i] = Math.min(ans[i], idx - i);
    }
    return ans;
};
