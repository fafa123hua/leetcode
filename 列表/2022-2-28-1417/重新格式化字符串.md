var reformat = function(s) {
    let str = s.replace(/[0-9]/g,'')
    let num = s.replace(/[a-z]/g,'')
    if(Math.abs(str.length - num.length) > 1) return ""
    let ans = ''
    if(str.length >= num.length){
        for(let i in str){
            ans = ans.concat(str[i], num[i] ? num[i] : '')
        }
    }else{
        for(let i in num){
            ans = ans.concat(num[i], str[i] ? str[i] : '')
        }
    }
    console.log(str,num)
    return ans
};
