var countGoodRectangles = function(rectangles) {
    let ans = 1
    let arr = []
    for(let i = 0; i < rectangles.length; i++){
        arr.push(Math.min(rectangles[i][1],rectangles[i][0]))
    }
    arr.sort((a,b) => b - a)
    for(let i = 1; i < arr.length; i++){
        if(arr[i] == arr[i - 1]){
            ans++
        }else{
            return ans
        }
    }
    return ans
};
