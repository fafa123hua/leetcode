```js
var checkZeroOnes = function(s) {
    let max1 = 0, max0 = 0
    let x = 0, y = 0
    for(let i of s){
        if(i == '1'){
            x++
            y = 0
            max1 = Math.max(max1, x)
        }else{
            x = 0
            y++
            max0 = Math.max(max0, y)
        }
    }
    return max1 > max0
};
```
