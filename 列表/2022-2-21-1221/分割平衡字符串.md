var balancedStringSplit = function(s) {
    let ans = 0
    let l = 0, r = 0
    for(let i = 0; i < s.length; i++){
        if(s[i] === 'L'){
            l++
        }else{
            r++
        }
        if(r == l){
            l = 0 
            r = 0
            ans++
        }
    }
    return ans
};
