var Solution = function(head) {
    this.head = head
};
Solution.prototype.getRandom = function() {
    let i = 1, ans = 0
    for (let node = this.head; node != null; node = node.next) {
        if (Math.floor(Math.random() * i) === 0) {
            ans = node.val;
        }
        ++i;
    }
    return ans
};
