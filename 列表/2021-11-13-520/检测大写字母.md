### 思路

除开第一个字母，剩下的大写字母左右到字母必须大写，注意当最后一个字母为大小字母时需要考虑边界问题。


```js
var detectCapitalUse = function(word) {
    if(word.length == 1){
        return true
    }
    for(let i = 1; i < word.length; i++){
        if(word[i].charCodeAt(0) - 97 < 0){
            if(word[i - 1].charCodeAt(0) - 97 >= 0){
                return false
            }
            if(i + 1 < word.length && word[i + 1].charCodeAt(0) - 97 >= 0){
                return false
            }
        }
    }
    return true
};
```
