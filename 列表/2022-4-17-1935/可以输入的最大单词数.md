var canBeTypedWords = function(text, brokenLetters) {
    let thash = text.split(' ')
    let borhash = new Set(brokenLetters.split(''))
    return thash.filter(item => {
        for(let i of item){
            if(borhash.has(i)){
                return false
            }
        }
        return true
    }).length
};
