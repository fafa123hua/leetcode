var isPalindrome = function(head) {
  if (head == null || head.next == null) {
    return true;
  }
  let f = head, s = head
  let pr
  while(f != null && f.next != null){
      pr = s
      s = s.next
      f = f.next.next
  }
  pr.next = null
  let head2 = null
  while(s){
      let next = s.next
      s.next = head2
      head2 = s
      s = next
  }
  while(head){
      if(head.val != head2.val){
          return false
      }else{
          head = head.next
          head2 = head2.next
      }
  }
  return true
};
