var minimumBuckets = function(street) {
    let t = -2
    let ans = 0
    for(let i = 0; i < street.length; i++){
        if(street[i] == 'H'){
            if(t == i - 1){
            }else if(i + 1 < street.length && street[i + 1] == '.'){
                ans++
                t = i + 1
            }else if(i > 0 && street[i - 1] == '.'){
                ans++
            }else{
                return -1
            }
        }
    }
    return ans
};
