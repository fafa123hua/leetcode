```js
var countConsistentStrings = function(allowed, words) {
    let hash = new Map()
    let ans = 0
    for(let i of allowed){
        hash.set(i, 1)
    }
    for(let i of words){
        let des = 1
        for(let j of i){
            if(!hash.has(j)){
                des = 0
                break
            }
        }
        if(des){
            ans++
        }
    }
    return ans
};
```
