var luckyNumbers  = function(matrix) {
    let ans = []
    let hash = new Map()
    for(let i of matrix){
        let min = Number.MAX_SAFE_INTEGER
        for(let j of i){
            min = Math.min(min, j)
        }
        hash.set(min, hash.get(min) + 1 || 1)
    }
    for(let i = 0; i < matrix[0].length; i++){
        let max = -1
        for(let j = 0; j < matrix.length; j++){
            max = Math.max(max, matrix[j][i])
        }
        if(hash.get(max)){
            ans.push(max)
        }
    }
    return ans
};
