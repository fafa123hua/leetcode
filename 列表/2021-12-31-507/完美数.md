```js
var checkPerfectNumber = function(num) {
    if(num == 1){
        return false
    }
    sum = 1
    for(let i = 2; i < Math.sqrt(num); i++){
        if(num % i == 0){
            sum += i
            if(i * i < num){
                sum += Math.floor(num / i)
            }
        }
    }
    return sum === num
};
```
