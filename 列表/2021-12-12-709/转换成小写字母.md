简单题重拳出击

```js
var toLowerCase = function(s) {
    let ans = []
    for(let i in s){
        if(s[i].charCodeAt(0) <= 90 && s[i].charCodeAt(0) >=65){
            ans.push(String.fromCharCode(s[i].charCodeAt(0)+32))
            continue
        }
        ans.push(s[i])
    }
    return ans.join('')
};
```
