var repeatedStringMatch = function(a, b) {
    let alen = a.length
    let blen = b.length
    let bool = new Array(26).fill(0)
    //判断是否有a不存在的字母
    for(let i= 0; i < alen; i++){
        bool[a[i].charCodeAt(0) - 'a'.charCodeAt(0)] = 1
    }
    for(let i =0; i < blen; i++){
        if(bool[b[i].charCodeAt(0) - 'a'] == 0){
            return -1
        }
    }
    let count = Math.ceil(blen / alen)
    if(a.repeat(count).includes(b))
        return count
    if(a.repeat(count+1).includes(b))
        return count + 1
    return -1
};
