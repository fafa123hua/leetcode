var checkValid = function(matrix) {
    let hash = new Map()
    for(let i = 0; i < matrix.length; i++){
        hash.clear()
        for(let j = 0; j < matrix.length ;j++){
            if(hash.has(matrix[i][j])){
                return false
            }
            hash.set(matrix[i][j],1)
        }
    }
    hash.clear()
    for(let i = 0; i < matrix.length; i++){
        hash.clear()
        for(let j = 0; j < matrix.length ;j++){
            if(hash.has(matrix[j][i])){
                return false
            }
            hash.set(matrix[j][i],1)
        }
    }
    return true
};
