var longestCommonPrefix = function(strs) {
    let ans = []
    let str1 = strs[0]
    for(let i = 0; i < strs[0].length; i++){
        let pan = 1
        for(let str of strs){
            if(str[i] != str1[i]){
                pan = 0
            }
        }
        if(pan){
            ans.push(str1[i])
        }else{
            break
        }
    }
    return ans.join('')
};
