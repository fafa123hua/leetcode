var minSubsequence = function(nums) {
    let all = 0
    let ans = []
    for(let i of nums){
        all += i
    }
    nums.sort((a,b) => b - a)
    let l = 0
    for(let i of nums){
        l += i
        all -= i
        ans.push(i)
        if(all < l){
            return ans 
        }
    }
};
