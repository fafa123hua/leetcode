### 代码

贪心思路，首先必须满足 `hand.length % groupSize == 0`

最小值一定在最前面，最小值又可以决定最大值，也就是说可以每次按最小值去找数组范围内的值，将其剔除然后继续查询

```js
var isNStraightHand = function(hand, groupSize) {
    let hash = new Map()
    if(groupSize == 1){
        return true
    }
    if(hand.length % groupSize != 0){
        return false
    }
    hand.sort((a, b) => a - b)
    for(let i of hand){
        hash.set(i, hash.get(i) + 1 || 1)
    }
    for(let i of hand){
        if(!hash.has(i)){
            continue
        }
        for(let j = 0; j < groupSize; j++){
            let num = i + j
            if(!hash.has(num)){
                return false
            }
            hash.set(num, hash.get(num) - 1);
            if (hash.get(num) == 0) {
                hash.delete(num);
            }
        }
    }
    return true
};
```
