var countQuadruplets = function(nums) {
    let ans = 0
    for(let i = 0; i < nums.length; i++){
        for(let j = i + 1; j < nums.length; j++){
            for(let z = j + 1; z < nums.length; z++){
                for(let x = z + 1; x < nums.length; x++){
                    if(nums[i] + nums[j] + nums[z] == nums[x]){
                        ans++
                    }
                }
            }
        }
    }
    return ans
};
