/**
 * @param {string} s
 * @return {string}
 */
var originalDigits = function(s) {
    let hash = new Map()
    for(let i in s){
        hash.set(s[i], hash.get(s[i]) + 1 || 1)
    }
    const cnt = new Array(10).fill(0);
    cnt[0] = hash.get('z') || 0;
    cnt[2] = hash.get('w') || 0;
    cnt[4] = hash.get('u') || 0;
    cnt[6] = hash.get('x') || 0;
    cnt[8] = hash.get('g') || 0;

    cnt[3] = (hash.get('h') || 0) - cnt[8];
    cnt[5] = (hash.get('f') || 0) - cnt[4];
    cnt[7] = (hash.get('s') || 0) - cnt[6];

    cnt[1] = (hash.get('o') || 0) - cnt[0] - cnt[2] - cnt[4];

    cnt[9] = (hash.get('i') || 0) - cnt[5] - cnt[6] - cnt[8];

    let ans = []
    for(let i = 0; i <= 9; i++){
        while(cnt[i]--){
            ans.push(String.fromCharCode(i + '0'.charCodeAt(0)))
        }
    }
    return ans.join('')
};
