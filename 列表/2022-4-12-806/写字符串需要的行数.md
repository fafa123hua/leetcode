var numberOfLines = function(widths, s) {
    let ans = []
    let h = 1
    let wi = 0
    for(let i = 0; i < s.length; i++){
        if(wi + widths[s[i].charCodeAt(0) - 'a'.charCodeAt(0)] > 100){
            wi = 0
            h++
        }
        wi += widths[s[i].charCodeAt(0) - 'a'.charCodeAt(0)]
    }
    ans.push(h, wi)
    return ans
};
