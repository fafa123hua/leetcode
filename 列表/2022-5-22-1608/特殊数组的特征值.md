var specialArray = function(nums) {
    nums.sort((a,b) => b - a)
    let ans = 0
    for(let i = 0; i <= nums[0]; i++){
        for(let j = 0; j < nums.length; j++){
            if(i <= nums[j]){
              ans++   
            }else{
                break
            }
        }
        if(ans == i){
            return ans
        }else{
            ans = 0
        }
    }
    return -1
};
