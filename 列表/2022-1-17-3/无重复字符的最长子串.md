var lengthOfLongestSubstring = function(s) {
    let l = -1
    let ans = 0
    let hash = new Set()
    for(let i = 0; i < s.length; i++){
        if(i != 0){
            hash.delete(s.charAt(i -1))
        }
        while(l + 1< s.length && !hash.has(s.charAt(l + 1))){
            hash.add(s.charAt(l + 1))
            l++
        }
        ans = Math.max(ans, l - i + 1)
    }
    return ans
};
