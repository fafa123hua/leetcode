var isSumEqual = function(firstWord, secondWord, targetWord) {
    return add(firstWord) + add(secondWord) == add(targetWord)
};
var add = function(str){
    let map = { a: 0, b: 1, c: 2, d: 3, e: 4, f: 5, g: 6, h: 7, i: 8, j: 9 }
    let ans = ''
    for(let i = str.length - 1; i >= 0; i--){
        ans = map[str[i]] + ans
    }
    return ans * 1
}
