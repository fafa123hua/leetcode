var findTheDifference = function(s, t) {
    let hash = new Map()
    for(let i of s){
        hash.set(i, hash.get(i) + 1 || 1)
    }
    for(let i of t){
        if(hash.has(i)){
            hash.set(i, hash.get(i) - 1)
            if(hash.get(i) < 0){
                return i
            }
        }else{
            return i
        }
    }
};
