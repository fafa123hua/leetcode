var missingRolls = function(rolls, mean, n) {
    let all = mean * (rolls.length + n)
    for(let i = 0; i < rolls.length; i++){
        all -= rolls[i]
    }
    if(all < n || all > 6 * n){
        return []
    }else{
        let ans = new Array(n).fill(Math.floor(all / n))
        for(let i = 0; i < all % n; i++){
            ans[i]++
        }
        return ans
    }
};
