var findErrorNums = function(nums) {
    let ans = new Map()
    for(let i of nums){
        ans.set(i, ans.get(i) + 1 || 1)
    }
    let a = 0, b = 0
    for(let i = 1; i <= nums.length; i++){
        if(ans.get(i)){
            if(ans.get(i) == 2){
                a = i
            }
        }else{
            b = i
        }
    }
    return [a, b]
};
