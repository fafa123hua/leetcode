var maximumTime = function(time) {
    let str = []
    if(time[0] == '?'){
        if(time[1] != '?' && time[1] > '3'){
            str.push(1)
        }else{
            str.push(2)
        }
    }else{
        str.push(time[0])
    }
    if(time[1] == '?'){
        if(str[0] != '2'){
            str.push(9)
        }else{
            str.push(3)
        }
    }else{
        str.push(time[1])
    }
    str.push(':')
    if(time[3] == '?'){
        str.push(5)
    }else{
        str.push(time[3])
    }
    if(time[4] == '?'){
        str.push(9)
    }else{
        str.push(time[4])
    }
    return str.join('')
};
