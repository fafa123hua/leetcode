```js
var findRelativeRanks = function(score) {
    const desc = ["Gold Medal", "Silver Medal", "Bronze Medal"];
    let hash = new Array(score.length).fill(0).map(()=> new Array(2).fill(0))
    for(let i in score){
        hash[i][0] = score[i]
        hash[i][1] = i
    }
    let ans = new Array(score.length).fill(0)
    hash.sort((a,b)=> b[0]-a[0])
    for(let i = 0; i < score.length; i++){
        if(i >= 3){
            ans[hash[i][1]] = '' + (i + 1)
        }else{
            ans[hash[i][1]] = desc[i]
        }
    }
    return ans
};
```
