var reversePrefix = function(word, ch) {
    let ans = []
    let f = -1
    for(let i = 0; i < word.length; i++){
        if(word[i] == ch){
            f = i
            break
        }
    }
    if(f == -1){
        return word
    }else{
        for(let i = f; i >= 0; i--){
            ans.push(word[i])
        }
        ans.push(word.slice(f + 1, word.length))
        return ans.join('')
    }
};
