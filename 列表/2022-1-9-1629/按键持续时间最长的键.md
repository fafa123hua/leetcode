var slowestKey = function(releaseTimes, keysPressed) {
    let char = []
    let hash = new Array(26).fill(0)
    let time = 0
    for(let i in releaseTimes){
        if(releaseTimes[i] - time > hash[keysPressed[i].charCodeAt(0) - 97]){
            hash[keysPressed[i].charCodeAt(0) - 97] = releaseTimes[i] - time
        }
        time = releaseTimes[i]
    }
    let max = -1
    for(let i = 0; i < 26; i++){
        if(hash[i]){
            if(max == -1){
                max = hash[i]
                char = String.fromCharCode(i + 97)
            }
            if(hash[i] >= max){
                char = String.fromCharCode(i + 97)
                max = hash[i]
            }
        }
    }
    return char
};
