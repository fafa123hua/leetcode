var goodDaysToRobBank = function(security, time) {
    let ans = []
    if(time >= security.length){
        return []
    }
    if(time == 0){
        for(let i in security){
            ans.push(i)
        }
        return ans
    }
    for(let i = time; i < security.length - time; i++){
        let pan = 1
        for(let j = i; j > i - time; j--){
            if(security[j - 1] < security[j]){
                pan = 0
                break
            }
        }
        for(let z = i; z < i + time; z++){
            if(security[z] > security[z + 1]){
                pan = 0
            }
        }
            
        if(pan){
            ans.push(i)
        }
    }
    return ans
};
