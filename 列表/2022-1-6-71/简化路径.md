```js
var simplifyPath = function(path) {
    let ans = []
    let newpath = path.split('/')
    for(let i of newpath){
        if(i == '..'){
            if(ans.length){
                ans.pop()
            }
        }else if(i.length && i != '.'){
            ans.push(i)
        }
    }
    return '/' + ans.join('/')
};
```
