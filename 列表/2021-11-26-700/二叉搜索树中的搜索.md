bst的规则

```js
 let dfs = function(root, val){
        if(!root){
            return null
        }
        if(root.val == val){
            return root
        }
        if(root.val > val){
            return dfs(root.left, val)
        }
        return dfs(root.right, val)
}
var searchBST = function(root, val) {
    return dfs(root, val)
};
```
