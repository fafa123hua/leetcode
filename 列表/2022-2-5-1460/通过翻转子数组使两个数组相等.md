var canBeEqual = function(target, arr) {
    let hash = new Map()
    for(let i = 0; i < target.length; i++){
        hash.set(target[i], hash.get(target[i]) + 1 || 1)
    }
    for(let i = 0; i < target.length; i++){
        if(!hash.get(arr[i]) || hash.get(arr[i] == 0)){
            return false
        }else{
            hash.set(arr[i], hash.get(arr[i]) - 1)
        }
    }
    return true
};
