var dominantIndex = function(nums) {
    if(nums.length == 1){
        return 0
    }
    let x = 0
    let y = -1
    for(let i = 1; i < nums.length; i++){
        if(nums[i] > nums[x]){
            y = x
            x = i
        }else if(y == -1 || nums[i] > nums[y]){
            y = i
        }
    }
     return nums[x] >= nums[y] * 2 ? x : -1;
};
