### 代码

建一个hash表，字母本身作为key值，统计magazine中字母出现的次数，在ransomNote中遍历一遍判断即可

```js
var canConstruct = function(ransomNote, magazine) {
    if(ransomNote.length > magazine.length){       //magazine < ransomNote肯定不行
        return false
    }
    let hash = new Map()
    for(let i of magazine){
        hash.set(i, hash.get(i) + 1 || 1)          //计数
    }
    for(let i of ransomNote){
        if(!hash.has(i)){                          //不存在则无法构成 ransomNote
            return false
        }else{
            hash.set(i, hash.get(i) - 1)
        }
        if(hash.get(i) < 0){
            return false
        }
    }
    return true
};
```
