var uniqueOccurrences = function(arr) {
    let pa = new Array(3000).fill(0)
    for(let i of arr){
        pa[i + 1001]++
    }
    pa.sort((a, b) => b - a)
    for(let i = 0; i < pa.length; i++){
        if(pa[i] > 0 && pa[i + 1] > 0){
            if(pa[i] == pa[i + 1]){
                return false
            }
        }else{
            break
        }
    }
    return true
};
