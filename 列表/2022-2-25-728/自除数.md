var selfDividingNumbers = function(left, right) {
    let ans = []
    for(let i = left; i <= right; i++){
        if(deft(i)){
            ans.push(i)
        }
    }
    return ans
};
var deft = function(x){
    let n = x
    while(n > 0){
        let set = n % 10
        n = Math.floor(n / 10)
        if(set == 0 || x % parseInt(set) != 0){
            return false
        }
    }
    return true
}
