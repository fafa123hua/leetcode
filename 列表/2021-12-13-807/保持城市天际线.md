简单题重拳出击

```js
var maxIncreaseKeepingSkyline = function(grid) {
    let n = grid.length
    let t = new Array(n).fill(0)
    let l = new Array(n).fill(0)
    let ans = 0
    for(let i = 0; i < n; i++){
        for(let j = 0; j < n; j++){
            t[i] = Math.max(t[i], grid[i][j])
            l[j] = Math.max(l[j], grid[i][j])

        }
    }
    for(let i = 0; i < n; i++){
        for(let j = 0; j < n; j++){
            ans += Math.min(t[i], l[j]) - grid[i][j]
        }
    }
    return ans
};
```
