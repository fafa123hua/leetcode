var construct2DArray = function(original, m, n) {
    let ans = new Array()
    let z = 0
    if(m * n != original.length){
        return []
    }
    for(let i = 0; i < m ; i++){
        let x = []
        for(let j = 0; j < n && z < original.length; j++,z++){
            x.push(original[z])
        }
        ans.push(x)
    }
    return ans
};
