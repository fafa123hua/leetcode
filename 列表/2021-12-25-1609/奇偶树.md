var isEvenOddTree = function(root) {
    let nodes = [root], level = 0
    while(nodes.length > 0){
        const nxt = new Array()
        let cur = level % 2 == 0 ? 0 : 1000001
        for(const node of nodes){
            if(level % 2 == node.val % 2 || (level % 2 == 0 && cur >= node.val) || (level % 2 != 0 && cur <= node.val))
            return false
            cur = node .val
            if(node.left != null)
                nxt.push(node.left)
            if(node.right != null)
                nxt.push(node.right)
        }
        nodes = nxt
        level++
    }
    return true
};
