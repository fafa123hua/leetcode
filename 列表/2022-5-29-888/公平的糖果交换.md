var fairCandySwap = function(aliceSizes, bobSizes) {
    let alla = 0
    let allb = 0
    for(let i of aliceSizes){
        alla += i
    }
    for(let i of bobSizes){
        allb += i
    }
    aliceSizes.sort((a, b) => a - b)
    bobSizes.sort((a, b) => a - b)
    let a = 0
    let b = 0
    let ans = (alla - allb) / 2
    while(a >= 0){
        let dpa = aliceSizes[a] - bobSizes[b]
        if(ans == dpa){
            return [aliceSizes[a], bobSizes[b]]
        }else if(ans > dpa){
            a++
        }else{
            b++
        }
    }
};
