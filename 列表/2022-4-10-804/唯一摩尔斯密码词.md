const MORSE = [".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."];
var uniqueMorseRepresentations = function(words) {
    let ans = new Set()
    for(let word of words){
        let str = ""
        for(let i of word){
            str += MORSE[i.charCodeAt() - 'a'.charCodeAt()]
        }
        ans.add(str)
    }
    return ans.size
}
