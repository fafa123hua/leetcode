```js
var minOperations = function(boxes) {
    let ans = []
    let left = 0, right = 0, num = 0 //记录球数和第一次操作数
    if(boxes[0] == '1') left++
    for(let i = 1; i < boxes.length; i++){
        if(boxes[i] == '1'){
            num += i
            right++
        }
    }
    ans.push(num)
    for(let i = 1; i < boxes.length; i++){
        num = num + left - right             //对于初始格子来说，每前进一格左边每有一个小球就会远一格，右边每有一个小球就会近一格
        if(boxes[i] == '1'){                 //更新左右球数
            left++
            right--
        }
        ans.push(num)
    }
    return ans
};
```
