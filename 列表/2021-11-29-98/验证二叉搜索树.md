var isValidBST = function(root) {
    let pre = []
    const dfs = function(root){
        if(!root){
            return
        }
        dfs(root.left);
        pre.push(root.val);
        dfs(root.right);
    }
    dfs(root)
    let x = pre[0]
    for(let i = 1; i < pre.length; i++){
        if(x >= pre[i]) return false
        x = pre[i]
    }
    
    return true
};
