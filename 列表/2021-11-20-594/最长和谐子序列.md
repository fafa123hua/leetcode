### **代码**

开心签到题,本质上你可以看作求数组中，**不同**且**相差为一**的数，它俩的出现次数的和

```js
var findLHS = function(nums) {
    let hash = new Map()
    let ans = 0
    for(let i of nums){
        if(hash.has(i)){                   //存起来计数
            hash.set(i, hash.get(i) + 1)
        }else{ 
            hash.set(i, 1)
        }
    }
    for(let key of hash.keys()){
        if(hash.has(key + 1)){                                     //判断，然后对比
            ans = Math.max(ans, hash.get(key) + hash.get(key + 1))
        }

    }
    return ans
};
```
