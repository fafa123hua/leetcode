var findMinDifference = function(timePoints) {
    let ans = Number.MAX_VALUE
    let time = []
    let n = timePoints.length
    if (n > 1440) {
        return 0;
    }
    timePoints.sort()
    for(let t of timePoints){
        time.push(((t[0].charCodeAt() - '0'.charCodeAt()) * 10 + (t[1].charCodeAt() - '0'.charCodeAt())) * 60 + (t[3].charCodeAt() - '0'.charCodeAt()) * 10 + (t[4].charCodeAt() - '0'.charCodeAt()))
    }
    for(let i = 0; i < time.length; i++){
        if(i + 1 < time.length && time[i + 1] - time[i] < ans){
            ans = time[i + 1] - time[i]
        }
    }
    ans = Math.min(ans, time[0] + 1440 - time[n - 1]);
    console.log(time)
    return ans
};
