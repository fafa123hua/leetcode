```js
var isPrefixOfWord = function(sentence, searchWord) {
    let str = sentence.split(" ")
    for(let i in str){
        let p = 1
        for(let j in searchWord){
            if(str[i][j] != searchWord[j]){
                p = 0
                break
            }
        }
        if(p){
            return parseInt(i) + 1
        }
        
    }
    return -1
};
```
