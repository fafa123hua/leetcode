var replaceDigits = function(s) {
    let ans = []
    for(let i = 0; i < s.length; i++){
        if(s[i] >= '0' && s[i] <= '9'){
           ans.push(String.fromCharCode(s[i-1].charCodeAt(0) + s[i].charCodeAt(0) - 48))
        }else{
            ans.push(s[i])
        }
    }
    return ans.join('')
};
