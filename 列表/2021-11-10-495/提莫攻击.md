### 模拟解决

每次进行判断看当前是否最长时间，重叠部分相减即可

```js
var findPoisonedDuration = function(timeSeries, duration) {
    let ans = 0, last = 0;
    for(let time of timeSeries){
        if(time >= last){
            ans += duration
        }else{
            ans += time + duration - last
        }
        last = time + duration
    }
    return ans
};
```
