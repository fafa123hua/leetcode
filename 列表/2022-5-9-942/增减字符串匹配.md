var diStringMatch = function(s) {
    let ans = []
    let min = 0
    let max = s.length
    for(let i of s){
        if(i == 'I'){
            ans.push(min)
            min++
        }else{
            ans.push(max)
            max--
        }
    }
    ans.push(min)
    return ans
};
