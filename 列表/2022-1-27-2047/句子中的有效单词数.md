var countValidWords = function(sentence) {
    let strs = sentence.split(' ')
    let ans = 0
    for(let i = 0; i < strs.length; i++){
        let pan = 1
        let x = 0
        for(let j = 0; j < strs[i].length; j++){
            if(strs[i][j] == '-'){
                x++
                if(x > 1){
                    pan = 0
                }
            }
            if((strs[i][j] == '!' || strs[i][j] == '.' || strs[i][j] == ',' ) && j != strs[i].length - 1){
                pan = 0
            }
            if(strs[i][j] == '-' && (((j <= 0 || j == strs[i].length - 1)) || strs[i][j - 1] < 'a' || strs[i][j - 1] > 'z' || strs[i][j + 1] > 'z' || strs[i][j + 1] < 'a')){
                pan = 0 
            }
            if(strs[i][j] != '!' && strs[i][j] != '.' && strs[i][j] != ',' && strs[i][j] != '-' && (strs[i][j] < 'a' || strs[i][j] > 'z')){
                pan = 0
            }
        }
        if(pan && strs[i].length > 0){
            ans++
        }
    }
    return ans
};
