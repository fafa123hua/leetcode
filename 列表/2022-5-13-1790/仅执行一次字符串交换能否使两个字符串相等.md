var areAlmostEqual = function(s1, s2) {
    if (s1 === s2) return true
    let diff = [], i = s1.length
    while (i--) {
        if (s1.charAt(i) !== s2.charAt(i)) {
            diff.push([s1.charAt(i), s2.charAt(i)])
        } 
    }
    return diff.length === 2 && diff[0][0] === diff[1][1] && diff[0][1] === diff[1][0]
};
