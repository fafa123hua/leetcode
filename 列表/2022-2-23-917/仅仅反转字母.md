var reverseOnlyLetters = function(s) {
    let l = 0, r = s.length - 1
    const arr = [...s]
    while(l < r){
        while(l < r && !(/[a-z]/ig.test(s[l]))){
            l++
        }
        while(l < r && !(/[a-z]/ig.test(s[r]))){
            r--
        }
        let t = arr[l]
        arr[l] = arr[r]
        arr[r] = t
        r--
        l++
    }
    return arr.join('')
};
