之后也补一个优先队列的

```js
var loudAndRich = function(richer, quiet) {
    var ans = new Array(quiet.length).fill(-1)
    var g = new Array(quiet.length).fill(0)
    for(let i = 0; i < quiet.length; i++){
        g[i] = []
    }
    for(let r of richer){
        g[r[1]].push(r[0])
    }
    for(let i = 0; i < quiet.length; i++){
        dfs(i, quiet, g, ans)
    }
    return ans
};
const dfs = function(x, quiet, g, ans){
    if(ans[x] !== -1){
        return
    }
    ans[x] = x
    for(let y of g[x]){
        dfs(y, quiet, g, ans)
        if(quiet[ans[y]] < quiet[ans[x]]){
            ans[x] = ans[y]
        }
    }
}
```
