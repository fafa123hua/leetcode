var smallerNumbersThanCurrent = function(nums) {
    let arr = new Array(101).fill(0)
    for(let i of nums){
        arr[i]++
    }
    for(let i = 1; i < 101; i++){
        arr[i] += arr[i - 1]
    }
    let ans = []
    for(let i of nums){
        ans.push(i ? arr[i - 1] : 0)
    }
    return ans
};
