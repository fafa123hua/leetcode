### 代码

```js
var numWaterBottles = function(numBottles, numExchange) {
    let ans = 0
    let drink = numBottles
    let bo = 0
    while(drink > 0){
        ans += drink                                     
        drink += bo                                    //喝完的加上之前余下的空瓶
        bo = drink % numExchange                       //看有没有多余空瓶
        drink = Math.floor((drink - bo) / numExchange) //去除多余空瓶后算能换多少
    }
    return ans
};
```
