基本dfs

```js
var findTilt = function(root) {
    let ans = 0
    var dfs = function(node){
        if(!node){
            return 0
        }
        let left = dfs(node.left)
        let right = dfs(node.right)
        ans += Math.abs(left - right)
        return left + right + node.val
    }
    dfs(root)
    return ans
};
```
