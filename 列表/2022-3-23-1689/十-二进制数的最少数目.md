```js
var prefixCount = function(words, pref) {
    let ans = 0
    for(let i = 0; i < words.length; i++){
        let de = 1
        for(let j = 0; j < pref.length; j++){
            if(words[i][j] != pref[j]){
                de = 0
                break
            }
        }
        if(de){
            ans++
        }
    }
    return ans
};
```
