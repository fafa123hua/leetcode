var maxNumberOfBalloons = function(text) {
    let ans = 0
    let hash = new Map()
    for(let i of text){
        hash.set(i, hash.get(i) + 1 || 1)
    }
    while(hash.get('b') > 0 && hash.get('a') > 0 && hash.get('n') > 0 && hash.get('l') > 1 &&hash.get('o') > 1){
        ans++
        hash.set('b', hash.get('b') - 1)
        hash.set('a', hash.get('a') - 1)
        hash.set('l', hash.get('l') - 2)
        hash.set('o', hash.get('o') - 2)
        hash.set('n', hash.get('n') - 1)
    }
    return ans
};
