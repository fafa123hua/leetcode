var largeGroupPositions = function(s) {
    let l = 0, r = 0, max = 0, tep = 0
    let ans = []
    while(r <= s.length - 1){
        if(s[l] == s[r]){
            max = r - l + 1
            r++
            if(r == s.length && max >= 3){
                ans.push([l, r - 1])
            }
        }else{
            if(max >= 3){
                ans.push([l, r - 1])
            }
            l = r
            max = 1
        }
        
    }
    return ans
};
