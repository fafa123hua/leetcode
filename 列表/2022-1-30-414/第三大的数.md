var thirdMax = function(nums) {
    nums.sort((a, b) => b - a)
    let pan = 1
    for(let i = 1; i < nums.length || pan == 3; i++){
        if(nums[i] != nums[i - 1]){
            pan++
            if(pan == 3){
                return nums[i]
            }
        }
    }
    return nums[0]
};
