var uncommonFromSentences = function(s1, s2) {
    let hash = new Map()
    let ans = new Array()
    let s1a = s1.split(' ')
    let s2a = s2.split(' ')
    for(let i of s1a){
        hash.set(i, hash.get(i) + 1 || 1)
    }
    for(let i of s2a){
        hash.set(i, hash.get(i) + 1 || 1)
    }
    console.log(hash)
    for(let i of hash.entries()){
        if(i[1] === 1){
            ans.push(i[0])
        }
    }
    return ans
};
