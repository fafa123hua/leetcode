var addToArrayForm = function(num, k) {
    let n = num.length - 1
    num.reverse()
    let i = 0
    let next = 0
    while(k){
        if(typeof num[i] === "number"){
            num[i] += k % 10 + next
            if(num[i] >= 10){
                next = 1
                num[i] -= 10
            }else{
                next = 0
            }
        }else{
            if(k % 10 + next >= 10){
                num.push(k % 10 + next - 10)
                next = 1
            }else{
                num.push(k % 10 + next)
                next = 0
            }
        }
        i++
        k = Math.floor(k / 10)
    }
    while(i <= n && next == 1){
        num[i] += next
        if(num[i] >= 10){
                next = 1
                num[i] -= 10
            }else{
                next = 0
        }
        i++
    }
    if(next == 1){
        num.push(1)
    }
    return num.reverse()
};
