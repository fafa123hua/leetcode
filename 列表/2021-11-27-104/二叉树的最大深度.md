```js
var maxDepth = function(root) {
    let ans = 0
    let x = 0
    var dfs = function(root){
        if(!root){
            return
        }
        x++
        dfs(root.left)
        dfs(root.right)
        ans = Math.max(x,ans)
        x--
    }
    dfs(root)
    return ans
};
```
