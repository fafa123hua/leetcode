```js
var countBalls = function(lowLimit, highLimit) {
    let ans = new Array(46).fill(0)
    for(let i = lowLimit;i <= highLimit;i++){
            let a = i;
            let sum = 0;
            while(a != 0){
                sum += a % 10;
                a = Math.floor(a / 10)
            }
            ans[sum]++;
        }
    ans.sort((a, b) => b - a)
    return ans[0]
};
```
