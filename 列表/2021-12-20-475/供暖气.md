var findRadius = function(houses, heaters) {
    houses.sort((a,b)=>a-b)
    heaters.sort((a,b)=>a-b)
    let ans = 0
    for(let i = 0,j = 0; i< houses.length; i++){
        let a = Math.abs(houses[i] - heaters[j])
        //大于
        while(heaters[j] <= houses[i] && j < heaters.length){
            a = houses[i] - heaters[j]
            j++
        }
        //小于
        if(j < heaters.length){
            a = Math.min(a, heaters[j] - houses[i])
        }
        ans = Math.max(a,ans)
        if(j > 0){
            j--
        }
    }
    return ans
};
