```js
var dayOfYear = function(date) {
    let ans = 0
    let y = +date.slice(0,4)
    let m = +date.slice(5,7)
    let d = +date.slice(8)
    const amount = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31];
    if (y % 400 === 0 || (y % 4 === 0 && y % 100 !== 0)) {
        ++amount[1];
    }
    for(let i = 0; i < m - 1; i++){
        ans += amount[i]
    }
    return ans + d
};
```
