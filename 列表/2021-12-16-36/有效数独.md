### 代码

注意数组范围就可以了

```js
var isValidSudoku = function(board) {
    let t = new Array(9).fill(0).map(() => new Array(9).fill(0))
    let l = new Array(9).fill(0).map(() => new Array(9).fill(0))
    let f = new Array(3).fill(0).map(() => new Array(3).fill(0).map(() => new Array(9).fill(0)))
    for(let i = 0; i < 9; i++){
        for(let j = 0; j < 9; j++){
            if(board[i][j] != '.'){
                let x = board[i][j].charCodeAt(0) - '0'.charCodeAt(0) - 1
                t[i][x]++
                l[j][x]++
                f[Math.floor(i / 3)][Math.floor(j / 3)][x]++
                if(t[i][x] > 1 || l[j][x] > 1 || f[Math.floor(i / 3)][Math.floor(j / 3)][x] > 1){
                    return false
                }
            }
        }
    }
    return true
};
```
