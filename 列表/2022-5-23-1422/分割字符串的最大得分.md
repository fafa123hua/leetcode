```js
var maxScore = function(s) {
    let l = 0, r = 0
    for(let i of s){
        if(i == '1'){
            l++
        }
    }
    let max = 0
    for(let i = 0; i < s.length - 1; i++){
        if(s[i] == '1'){
            l--
        }else{
            r++
        }
        max = Math.max(max, r + l)
    }
    return max
};
```
